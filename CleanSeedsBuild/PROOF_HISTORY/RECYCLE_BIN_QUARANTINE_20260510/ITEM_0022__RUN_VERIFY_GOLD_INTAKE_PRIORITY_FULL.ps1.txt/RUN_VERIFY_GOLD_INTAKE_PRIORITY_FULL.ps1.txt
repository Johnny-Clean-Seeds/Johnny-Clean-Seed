function Add-Line($Text) {
    $Lines.Add($Text) | Out-Null
    Write-Host $Text
}

function Count-Marker($Text, $Marker) {
    return ([regex]::Matches($Text, [regex]::Escape($Marker))).Count
}

function Require-Path($Path) {
    if (-not (Test-Path -LiteralPath $Path)) {
        Add-Line "FAIL missing required path: $Path"
        $script:Fail = $true
    } else {
        Add-Line "PASS path exists: $Path"
    }
}

function Add-Block-If-Missing($Path, $Marker, $Block, $Label) {
    $Content = Get-Content -LiteralPath $Path -Raw
    if ($Content -notlike "*$Marker*") {
        Add-Content -LiteralPath $Path -Value "`r`n$Block" -Encoding UTF8
        Add-Line "PASS $Label added"
    } else {
        Add-Line "PASS $Label already had marker"
    }
}

function Check-One($Label, $Text, $Marker) {
    $Count = Count-Marker $Text $Marker
    if ($Count -eq 1) {
        Add-Line "PASS $Label marker exactly once"
    } else {
        Add-Line "FAIL $Label marker count: $Count"
        $script:Fail = $true
    }
}
